class _compilState extends State<compil>{

  String mage  = 'Not Found' ;

  @override
  void initState(){
    super.initState();}
  @override
  Widget build(BuildContext context){
  return Scaffold(
      appBar:AppBar(backgroundColor:Colors.black),
      body:Padding(padding:EdgeInsets.all(8),
